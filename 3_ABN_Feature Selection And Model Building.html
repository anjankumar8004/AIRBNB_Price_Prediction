RMSE=[]
def modelselect(train_X,train_Y,test_X,test_Y):
    lm=LinearRegression()
    dtree=DecisionTreeRegressor(random_state=45)
    rf=RandomForestRegressor(random_state=45)
    nn=MLPRegressor(hidden_layer_sizes=(5,10,5),random_state=45)
    xgb=xgboost.XGBRegressor(random_state=45,objective='reg:linear',learning_rate=0.1,subsample=0.8,colsample_bytree=0.8,colsample_bylevel=0.8, tree_method='gpu_hist', predictor='gpu_predictor')
    
    lm.fit(train_X,train_Y)
    e=mean_squared_error(lm.predict(test_X),test_Y,squared=False)
    RMSE.append({"LinearModel":e})
    print("-----LinearModel Completed------",str(e))
    
    dtree.fit(train_X,train_Y)
    e=mean_squared_error(dtree.predict(test_X),test_Y,squared=False)
    RMSE.append({"DecisionTree":e})
    print("-------DecisionTree Completed-----",str(e))
    
    rf.fit(train_X,train_Y)
    e=mean_squared_error(rf.predict(test_X),test_Y,squared=False)
    RMSE.append({"RandomForest":e})
    print("-----RandomForest Completed------",str(e))
    
    # Fit only to the training data
    scaler.fit(train_X)
    scaledTrainX=scaler.transform(train_X)
    scaledTestX=scaler.transform(test_X)
    nn.fit(scaledTrainX,train_Y)
    e=mean_squared_error(nn.predict(scaledTestX),test_Y,squared=False)
    RMSE.append({"Neural Networks":e})
    print("-----Neural Networks Completed------",str(e))
    
    xgb.fit(train_X,train_Y)
    e=mean_squared_error(xgb.predict(test_X),test_Y,squared=False)
    RMSE.append({"XGBoost":e})
    print("------XGBoost Completed-------",str(e))
